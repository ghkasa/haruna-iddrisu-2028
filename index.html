        if (!timelineContainer || !timelineDetailsContainer) return;

                <div class="timeline-dot w-6 h-6 rounded-full bg-[var(--ndc-green)] mx-auto transition-all duration-300 border-4 border-slate-100"></div>

                <p class="timeline-title font-semibold text-slate-600 mt-2">${item.title}</p>

            </div>

        `).join('');


        const timelineItems = document.querySelectorAll('.timeline-item');

        

        function showTimelineDetails(id) {

            const item = appData.timeline.find(d => d.id == id);

            if (!item) return;


            timelineDetailsContainer.innerHTML = `

                <h3 class="font-display text-2xl text-[var(--ndc-green)]">${item.title} <span class="font-accent text-lg text-slate-500 ml-2">(${item.duration})</span></h3>

                <p class="my-4 text-slate-700">${item.details}</p>

                <ul class="space-y-2 list-disc list-inside text-slate-600 marker:text-[var(--ndc-red)]">

                    ${item.actions.map(action => `<li>${action}</li>`).join('')}

                </ul>

            `;

            timelineItems.forEach(el => el.classList.toggle('active', el.dataset.id == id));

        }


        timelineItems.forEach(item => {

            item.addEventListener('click', () => showTimelineDetails(item.dataset.id));

        });


        showTimelineDetails(1);

    }
